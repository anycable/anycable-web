# Broadcast from FastAPI via HTTP API
import httpx

@app.post("/messages")
async def create_message(msg: Message):
    async with httpx.AsyncClient() as client:
        await client.post(
            f"{ANYCABLE_URL}/api/v1/broadcasts",
            json={
                "stream": f"chat/{msg.room_id}",
                "data": msg.json()
            }
        )
    return {"ok": True}
